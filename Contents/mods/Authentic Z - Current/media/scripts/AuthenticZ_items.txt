  	 recipe Insert Battery into Military Flashlight
   {
     destroy Authentic_MilitaryFlashlightGrey,
     destroy Battery,

     Result:Authentic_MilitaryFlashlightGrey,
     Time:30,
     OnTest:TorchBatteryInsert_TestIsValid,
     OnCreate:TorchBatteryInsert_OnCreate,
      Prop2:Authentic_MilitaryFlashlightGrey,
